def gcd(m,n):
    cf=[]
    for i in range(1,min(m,n)+1):
        if (m%i)==0 and (n%1)==0:
            cf.append(i)

def factors(n):
    fl=[]

    for i in range(1,n+1):
        if n%i==0:
            fl.append(i)
    return fl

print(factors(8989))

def isprime(n):
    return(factors(n)==[1,n])

def prime_product(n):
    for i in range(1,n+1):
        if n%i==0:
            if isprime(i) and isprime(n//i):
                return(True)
    return(False)

n=int(input())

def del_char(s,c):
    if len(c)!=1:
        return s
    snew=""
    for i in s:
        if i!=c:
            snew=snew+i
    return snew


s=input()
c=input()

print(del_char(s,c))

def shuffle(l1,l2):
    if len(l1)<len(l2):
        minlen=len(l1)
    else:
        minlen=len(l2)

    shuffled=[]
    for i in range(minlen):
        shuffled.append(l1[i])
        shuffled.append(l2[i])

    shuffled=shuffled+l1[minlen:] + l2[minlen:]

    return shuffled

l1=eval(input())
l2=eval(input())
print(shuffle(l1,l2))
